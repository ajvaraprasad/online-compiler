   145тЖТ   145тЖТ// тФАтФАтФА Register error link provider on xterm instance тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   146тЖТ   146тЖТ// Uses xterm.js v6 registerLinkProvider API to make detected error patterns
   147тЖТ   147тЖТ// clickable. When a line contains a recognizable error format (Python, C/C++,
   148тЖТ   148тЖТ// Java, Node.js), the matching region becomes a link with underline styling
   149тЖТ   149тЖТ// and a pointer cursor. Clicking navigates the Monaco editor to the error line.
   150тЖТ   150тЖТ
   151тЖТ   151тЖТfunction registerErrorLinkProvider(xterm: XTerm) {
   152тЖТ   152тЖТ  const linkProvider: ILinkProvider = {
   153тЖТ   153тЖТ    provideLinks(bufferLineNumber: number, callback: (links: ILink[] | undefined) => void): void {
   154тЖТ   154тЖТ      try {
   155тЖТ   155тЖТ        // Get the buffer line at the given row
   156тЖТ   156тЖТ        const bufferLine = xterm.buffer.active.getLine(bufferLineNumber);
   157тЖТ   157тЖТ        if (!bufferLine) {
   158тЖТ   158тЖТ          callback(undefined);
   159тЖТ   159тЖТ          return;
   160тЖТ   160тЖТ        }
   161тЖТ   161тЖТ
   162тЖТ   162тЖТ        // Translate the buffer line to a plain string (visible text only)
   163тЖТ   163тЖТ        const lineText = bufferLine.translateToString(true);
   164тЖТ   164тЖТ        if (!lineText || lineText.trim().length === 0) {
   165тЖТ   165тЖТ          callback(undefined);
   166тЖТ   166тЖТ          return;
   167тЖТ   167тЖТ        }
   168тЖТ   168тЖТ
   169тЖТ   169тЖТ        // Quick pre-filter: skip lines that don't look like errors
   170тЖТ   170тЖТ        if (!isPotentialErrorLine(lineText)) {
   171тЖТ   171тЖТ          callback(undefined);
   172тЖТ   172тЖТ          return;
   173тЖТ   173тЖТ        }
   174тЖТ   174тЖТ
   175тЖТ   175тЖТ        // Strip ANSI codes for parsing (the visible text from translateToString
   176тЖТ   176тЖТ        // should already be clean, but be defensive)
   177тЖТ   177тЖТ        const cleanText = stripAnsi(lineText);
   178тЖТ   178тЖТ
   179тЖТ   179тЖТ        // Parse for error patterns
   180тЖТ   180тЖТ        const parsed = parseTerminalError(cleanText);
   181тЖТ   181тЖТ        if (!parsed) {
   182тЖТ   182тЖТ          callback(undefined);
   183тЖТ   183тЖТ          return;
   184тЖТ   184тЖТ        }
   185тЖТ   185тЖТ
   186тЖТ   186тЖТ        // Calculate the range for the clickable region.
   187тЖТ   187тЖТ        // xterm positions are 1-based. matchStart/matchEnd are 0-based string indices.
   188тЖТ   188тЖТ        const startX = parsed.matchStart + 1;  // Convert 0-based тЖТ 1-based
   189тЖТ   189тЖТ        const endX = Math.min(parsed.matchEnd + 1, lineText.length); // Inclusive end тЖТ +1, clamp
   190тЖТ   190тЖТ        const y = bufferLineNumber;
   191тЖТ   191тЖТ
   192тЖТ   192тЖТ        const range: IBufferRange = {
   193тЖТ   193тЖТ          start: { x: startX, y },
   194тЖТ   194тЖТ          end: { x: endX, y },
   195тЖТ   195тЖТ        };
   196тЖТ   196тЖТ
   197тЖТ   197тЖТ        const linkText = cleanText.substring(parsed.matchStart, parsed.matchEnd);
   198тЖТ   198тЖТ
   199тЖТ   199тЖТ        const link: ILink = {
   200тЖТ   200тЖТ          range,
   201тЖТ   201тЖТ          text: linkText,
   202тЖТ   202тЖТ          decorations: {
   203тЖТ   203тЖТ            pointerCursor: true,
   204тЖТ   204тЖТ            underline: true,
   205тЖТ   205тЖТ          },
   206тЖТ   206тЖТ          activate(_event: MouseEvent, _text: string): void {
   207тЖТ   207тЖТ            navigateEditorToLine(parsed.line, parsed.column);
   208тЖТ   208тЖТ          },
   209тЖТ   209тЖТ          hover(event: MouseEvent, _text: string): void {
   210тЖТ   210тЖТ            // Show tooltip near the mouse position
   211тЖТ   211тЖТ            const terminalEl = xterm.element;
   212тЖТ   212тЖТ            if (!terminalEl) return;
   213тЖТ   213тЖТ
   214тЖТ   214тЖТ            const tooltip = getOrCreateTooltip(terminalEl);
   215тЖТ   215тЖТ            tooltip.textContent = `Click to go to line ${parsed.line}${parsed.column ? `:${parsed.column}` : ''}`;
   216тЖТ   216тЖТ            tooltip.style.display = 'block';
   217тЖТ   217тЖТ
   218тЖТ   218тЖТ            // Position relative to the terminal element
   219тЖТ   219тЖТ            const termRect = terminalEl.getBoundingClientRect();
   220тЖТ   220тЖТ            const offsetX = event.clientX - termRect.left + 12;
   221тЖТ   221тЖТ            const offsetY = event.clientY - termRect.top - 28;
   222тЖТ   222тЖТ            tooltip.style.left = `${offsetX}px`;
   223тЖТ   223тЖТ            tooltip.style.top = `${offsetY}px`;
   224тЖТ   224тЖТ          },
   225тЖТ   225тЖТ          leave(): void {
   226тЖТ   226тЖТ            hideTooltip();
   227тЖТ   227тЖТ          },
   228тЖТ   228тЖТ          dispose(): void {
   229тЖТ   229тЖТ            hideTooltip();
   230тЖТ   230тЖТ          },
   231тЖТ   231тЖТ        };
   232тЖТ   232тЖТ
   233тЖТ   233тЖТ        callback([link]);
   234тЖТ   234тЖТ      } catch (err) {
   235тЖТ   235тЖТ        // Non-blocking: never let link parsing break terminal rendering
   236тЖТ   236тЖТ        console.warn('[Terminal] Link provider error:', err);
   237тЖТ   237тЖТ        callback(undefined);
   238тЖТ   238тЖТ      }
   239тЖТ   239тЖТ    },
   240тЖТ   240тЖТ  };
   241тЖТ   241тЖТ
   242тЖТ   242тЖТ  xterm.registerLinkProvider(linkProvider);
   243тЖТ   243тЖТ  console.log('[Terminal] Error link provider registered');
   244тЖТ   244тЖТ}
