    82тЖТfunction formatPhaseData(phase: string, data: Record<string, unknown>): string {
    83тЖТ  const parts: string[] = [];
    84тЖТ  switch (phase) {
    85тЖТ    case 'lexical_analysis':
    86тЖТ      if (data.totalTokens) parts.push(`${data.totalTokens} tokens`);
    87тЖТ      if (data.linesOfCode) parts.push(`${data.linesOfCode} LOC`);
    88тЖТ      break;
    89тЖТ    case 'parsing':
    90тЖТ      if (data.totalNodes) parts.push(`${data.totalNodes} nodes`);
    91тЖТ      if (data.functionCount) parts.push(`${data.functionCount} functions`);
    92тЖТ      break;
    93тЖТ    case 'semantic_analysis':
    94тЖТ      if (data.totalSymbols) parts.push(`${data.totalSymbols} symbols`);
    95тЖТ      break;
    96тЖТ    case 'ir_generation':
    97тЖТ      if (data.totalInstructions) parts.push(`${data.totalInstructions} instructions`);
    98тЖТ      break;
    99тЖТ    case 'optimization':
   100тЖТ      if (data.instructionsBefore && data.instructionsAfter) {
   101тЖТ        parts.push(`${data.instructionsBefore} -> ${data.instructionsAfter} instructions`);
   102тЖТ      }
   103тЖТ      break;
   104тЖТ    case 'security_analysis':
   105тЖТ      if (data.riskLevel) parts.push(`risk: ${data.riskLevel}`);
   106тЖТ      break;
   107тЖТ    case 'code_generation':
   108тЖТ      if (data.mode) {
   109тЖТ        const modeInfo = MODE_DISPLAY[data.mode as string];
   110тЖТ        parts.push(modeInfo?.name || String(data.mode));
   111тЖТ      }
   112тЖТ      if (data.linesGenerated) parts.push(`${data.linesGenerated} lines`);
   113тЖТ      break;
   114тЖТ    case 'compilation':
   115тЖТ      break;
   116тЖТ    case 'execution':
   117тЖТ      if (data.mode) {
   118тЖТ        const modeInfo = MODE_DISPLAY[data.mode as string];
   119тЖТ        parts.push(modeInfo?.name || String(data.mode));
   120тЖТ      }
   121тЖТ      break;
   122тЖТ  }
   123тЖТ  return parts.length > 0 ? ` (${parts.join(', ')})` : '';
   124тЖТ}

   236тЖТ    setCurrentRequestId(requestId);
   237тЖТ    currentRequestIdRef.current = requestId;
   238тЖТ
   239тЖТ    // Wait for the terminal to be visible and the xterm renderer to complete
   240тЖТ    // a fit+refresh cycle before writing any data.  This eliminates the race
   241тЖТ    // condition where output is written while the container still has zero
   242тЖТ    // dimensions after transitioning from display:none / visibility:hidden.
   243тЖТ    //
   244тЖТ    // The terminalReady promise is resolved inside Terminal.tsx after:
   245тЖТ    //   1. The container becomes visible (CSS transition completes)
   246тЖТ    //   2. fitAddon.fit() succeeds (dimensions are non-zero)
   247тЖТ    //   3. terminal.refresh() forces a full repaint
   248тЖТ    //
   249тЖТ    // Fallback: if the promise takes too long (e.g. terminal was already
   250тЖТ    // visible), the 200ms timeout ensures we don't hang forever.
   251тЖТ    const terminalReadyWithTimeout = Promise.race([
   252тЖТ      waitForTerminalReady(),
   253тЖТ      new Promise<void>((resolve) => setTimeout(resolve, 200)),
   254тЖТ    ]);
   255тЖТ
   256тЖТ    terminalReadyWithTimeout.then(() => {
   257тЖТ      const store = useIDEStore.getState();
   258тЖТ      if (store.settings.terminalClearOnRun) {
   259тЖТ        store.clearTerminal();

   261тЖТ
   262тЖТ      // Check if the terminal service is reachable
   263тЖТ      if (!isConnected()) {
   264тЖТ        store.writeToTerminal(
   265тЖТ          ANSI.YELLOW + '[WARN]  Terminal service not connected. Attempting to connect...' + ANSI.RESET + '\r\n'
   266тЖТ        );
   267тЖТ      }
   268тЖТ
   269тЖТ      // Define the WebSocket event handler тАФ handles pipeline events from the compiler engine
   270тЖТ      const handleEvent = (event: WSEvent) => {
   271тЖТ        const currentState = useIDEStore.getState();
   272тЖТ        // Ignore events for a different requestId
   273тЖТ        if (currentState.currentRequestId !== requestId) return;
   274тЖТ
   275тЖТ        switch (event.type) {
   276тЖТ          case 'start': {
   277тЖТ            // Already set executing=true above
   278тЖТ            break;
   279тЖТ          }
   280тЖТ          case 'phase': {
   281тЖТ            // Pipeline phase transitions from the compiler engine
   282тЖТ            try {
   283тЖТ              const phaseData = JSON.parse(event.data || '{}');
   284тЖТ              const phaseName = phaseData.phase;
   285тЖТ              const display = PHASE_DISPLAY[phaseName] || { name: phaseName };
   286тЖТ              const phaseInfo: CompilerPhaseInfo = {
   287тЖТ                phase: phaseName,
   288тЖТ                status: phaseData.status,
   289тЖТ                durationMs: phaseData.durationMs,
   290тЖТ                data: phaseData.data,
   291тЖТ              };
   292тЖТ              currentState.updatePipelinePhase(phaseInfo);
   293тЖТ
   294тЖТ              if (phaseData.status === 'running') {
   295тЖТ                // [INFO] Phase Name...
   296тЖТ                currentState.writeToTerminal(
   297тЖТ                  ANSI.GREEN + `[INFO]  ${display.name}...` + ANSI.RESET + '\r\n'
   298тЖТ                );
   299тЖТ              } else if (phaseData.status === 'completed') {
   300тЖТ                const dataStr = phaseData.data ? formatPhaseData(phaseName, phaseData.data) : '';
   301тЖТ                if (phaseData.warning) {
   302тЖТ                  // [WARN] Phase completed with warnings
   303тЖТ                  currentState.writeToTerminal(
   304тЖТ                    ANSI.YELLOW + `[WARN]  ${display.name}${dataStr}` + ANSI.RESET + '\r\n'
   305тЖТ                  );
   306тЖТ                } else {
   307тЖТ                  // [OK] Phase completed successfully
   308тЖТ                  currentState.writeToTerminal(
   309тЖТ                    ANSI.DIM + `[OK]    ${display.name}${dataStr}` + ANSI.RESET + '\r\n'
   310тЖТ                  );
   311тЖТ                }
   312тЖТ              } else if (phaseData.status === 'failed') {
   313тЖТ                const dataStr = phaseData.data ? formatPhaseData(phaseName, phaseData.data) : '';
   314тЖТ                if (phaseData.fatal) {
   315тЖТ                  // [FATAL] Phase failed тАФ pipeline terminates
   316тЖТ                  currentState.writeToTerminal(
   317тЖТ                    ANSI.RED + ANSI.BOLD + `[FATAL] ${display.name} failed${dataStr}` + ANSI.RESET + '\r\n'
   318тЖТ                  );
   319тЖТ                } else {
   320тЖТ                  // [ERROR] Phase failed (non-fatal)
   321тЖТ                  currentState.writeToTerminal(
   322тЖТ                    ANSI.RED + `[ERROR] ${display.name} failed${dataStr}` + ANSI.RESET + '\r\n'
   323тЖТ                  );
   324тЖТ                }
   325тЖТ              } else if (phaseData.status === 'skipped') {
   326тЖТ                const reason = phaseData.message ? ` -- ${phaseData.message}` : '';
   327тЖТ                // [SKIP] Phase -- reason
   328тЖТ                currentState.writeToTerminal(
   329тЖТ                  ANSI.DIM + `[SKIP]  ${display.name}${reason}` + ANSI.RESET + '\r\n'
   330тЖТ                );
   331тЖТ              }
   332тЖТ            } catch {}
   333тЖТ            break;
   334тЖТ          }
   335тЖТ          case 'stdout': {
   336тЖТ            // PTY output includes echo + program output тАФ just display it
   337тЖТ            currentState.writeToTerminal(event.data || '');
   338тЖТ            break;
   339тЖТ          }
   340тЖТ          case 'stderr': {
   341тЖТ            // Check if this is a pipeline diagnostic (already has ANSI codes)
   342тЖТ            const data = event.data || '';
   343тЖТ            if (data.includes('[ERROR]') || data.includes('[WARN]') || data.includes('[FATAL]') || data.includes('Compilation terminated.')) {
   344тЖТ              // Pipeline diagnostic messages тАФ display as-is (already have ANSI codes)
   345тЖТ              currentState.writeToTerminal(data);
   346тЖТ            } else {
   347тЖТ              // Regular stderr from program тАФ display in red
   348тЖТ              currentState.writeToTerminal(ANSI.RED + data + ANSI.RESET);
   349тЖТ            }
   350тЖТ            break;
   351тЖТ          }
   352тЖТ          case 'killed': {
   353тЖТ            currentState.writeToTerminal('\r\n' + ANSI.YELLOW + 'Execution killed' + ANSI.RESET + '\r\n');
   354тЖТ            break;
   355тЖТ          }
   356тЖТ          case 'end': {
   357тЖТ            currentState.setExecuting(false);
   358тЖТ
   359тЖТ            const exitCode = event.exitCode;
   360тЖТ            const executionTime = event.executionTime || 0;
   361тЖТ            const timeStr = (executionTime / 1000).toFixed(2);
   362тЖТ
   363тЖТ            // Parse summary if available
   364тЖТ            let summary: Record<string, any> | undefined;
   365тЖТ            try {
   366тЖТ              const endPayload = JSON.parse(event.data || '{}');
   367тЖТ              summary = endPayload.summary;
   368тЖТ              // Store metrics in pipeline state
   369тЖТ              if (summary?.metrics) {
   370тЖТ                currentState.setPipelineMetrics(summary.metrics);
   371тЖТ              }
   372тЖТ            } catch {}
   373тЖТ
   374тЖТ            currentState.writeToTerminal('\r\n');
   375тЖТ
   376тЖТ            if (exitCode === 0) {
   377тЖТ              // VS Code style: clean exit status
   378тЖТ              currentState.writeToTerminal(
   379тЖТ                ANSI.DIM + `Process exited with code 0 in ${timeStr}s` + ANSI.RESET + '\r\n'
   380тЖТ              );
   381тЖТ            } else if (exitCode === -1) {
   382тЖТ              currentState.writeToTerminal(
   383тЖТ                ANSI.RED + `Process timed out after ${timeStr}s` + ANSI.RESET + '\r\n'
   384тЖТ              );
   385тЖТ            } else {
   386тЖТ              // Check if it was blocked by a pipeline phase
   387тЖТ              const phases = currentState.pipelineState.phases;
   388тЖТ              const failedPhase = phases.find(p => p.status === 'failed');
   389тЖТ              if (failedPhase) {
   390тЖТ                const display = PHASE_DISPLAY[failedPhase.phase] || { name: failedPhase.phase };
   391тЖТ                currentState.writeToTerminal(
   392тЖТ                  ANSI.RED + `Execution blocked: ${display.name} failed` + ANSI.RESET + '\r\n'
   393тЖТ                );
   394тЖТ              } else {
   395тЖТ                currentState.writeToTerminal(
   396тЖТ                  ANSI.RED + `Process exited with code ${exitCode} in ${timeStr}s` + ANSI.RESET + '\r\n'
   397тЖТ                );
   398тЖТ              }
   399тЖТ            }
   400тЖТ
   401тЖТ            if (currentRequestIdRef.current === requestId) {
   402тЖТ              currentState.setCurrentRequestId(null);
   403тЖТ              currentRequestIdRef.current = null;
   404тЖТ            }
   405тЖТ            break;
   406тЖТ          }
   407тЖТ        }
   408тЖТ      };
   409тЖТ
   410тЖТ      // Start execution via WebSocket -> PTY on the backend.
   411тЖТ      executeCode(
   412тЖТ        activeTab.content,
   413тЖТ        language,
   414тЖТ        requestId,
   415тЖТ        undefined, // interactive: all input via terminal
   416тЖТ        true,
   417тЖТ        handleEvent
   418тЖТ      );
   419тЖТ    }); // Wait for terminal ready
   420тЖТ  }, [activeTab, isExecuting, language, setTerminalOpen, setExecuting, setCurrentRequestId, resetPipeline, updatePipelinePhase, setPipelineMetrics]);
   421тЖТ
   422тЖТ  const handleStop = useCallback(async () => {
   423тЖТ    const state = useIDEStore.getState();
   424тЖТ    const requestId = state.currentRequestId || currentRequestIdRef.current;
   425тЖТ
   426тЖТ    if (requestId) {

   428тЖТ    }
   429тЖТ

   431тЖТ    setCurrentRequestId(null);
   432тЖТ    currentRequestIdRef.current = null;
   433тЖТ    writeToTerminal(ANSI.YELLOW + 'Execution stopped by user' + ANSI.RESET + '\r\n');
   434тЖТ  }, [setExecuting, setCurrentRequestId, writeToTerminal]);
   435тЖТ
   436тЖТ  const handleDownload = () => {
   437тЖТ    if (!activeTab) return;
   438тЖТ
   439тЖТ    const blob = new Blob([activeTab.content], { type: 'text/plain' });
   440тЖТ    const url = URL.createObjectURL(blob);

   443тЖТ    a.download = activeTab.name;
   444тЖТ    document.body.appendChild(a);
   445тЖТ    a.click();
   446тЖТ    document.body.removeChild(a);
   447тЖТ    URL.revokeObjectURL(url);
   448тЖТ  };
   449тЖТ
   450тЖТ  const handleSave = async () => {
   451тЖТ    if (!activeTab || !isAuthenticated) return;
   452тЖТ
   453тЖТ    setIsSaving(true);
   454тЖТ    try {

   461тЖТ      } else {
   462тЖТ        const data = await filesAPI.create(activeTab.name, activeTab.language, activeTab.content);
   463тЖТ        setRemoteFiles([...remoteFiles, data.file]);
   464тЖТ      }
   465тЖТ      markTabClean(activeTab.id);
   466тЖТ    } catch (err: any) {
   467тЖТ      console.error('Save failed:', err.message);
   468тЖТ    } finally {
